\chapter{Methodology}
\label{ch:methodology}

This chapter presents the cooperative striker--jammer problem, the two
simulation environments used in this work, the baseline flattened MAPPO
algorithm, and the proposed FOFE-MAPPO variant. 

\section{Problem Formulation}
\label{sec:problem}

The cooperative strike--EA engagement is modelled as a decentralised
partially-observable Markov decision process (Dec-POMDP)
$\langle \mathcal{N}, \mathcal{S}, \{\mathcal{A}_i\},
\{\mathcal{O}_i\}, P, \mathcal{R}, \gamma\rangle$ on a bounded two-dimensional
plane $\mathcal{W}=[0,1]^2$ with reflecting boundaries. The agent set is
partitioned by role into strikers and jammers,
\begin{equation}
\mathcal{N} \;=\; \mathcal{N}_{\mathrm{s}} \cup \mathcal{N}_{\mathrm{j}},
\qquad
|\mathcal{N}_{\mathrm{s}}|=N_{\mathrm{s}},\;
|\mathcal{N}_{\mathrm{j}}|=N_{\mathrm{j}},
\end{equation}
operating in an environment that additionally contains a set of radars
$\mathcal{R}=\{1,\dots,N_{\mathrm{r}}\}$ and protected targets
$\mathcal{T}=\{1,\dots,N_{\mathrm{t}}\}$. The team objective is to destroy
$\mathcal{T}$; jammers degrade radar coverage so that strikers may penetrate
the defended airspace. Training follows the centralised-training /
decentralised-execution (CTDE) paradigm: during execution each agent conditions
only on its local observation $o_i$, while during training the critics have
access to the global state $s\in\mathcal{S}$.

\section{Agent Kinematics}
\label{sec:kinematics}

Each agent $i\in\mathcal{N}$ follows a planar unicycle model with
double-integrator heading dynamics. The continuous agent state is
$\mathbf{s}_i=[x_i,y_i,\psi_i,v_i,\dot{\psi}_i]^{\top}$, giving position,
heading, linear speed and heading rate. One environment step integrates the
commanded acceleration $a_i^{v}\in[-1,1]$ and angular-acceleration multiplier
$a_i^{\omega}\in[-1,1]$ through
\begin{align}
v_{i,t+1} \;&=\; \mathrm{clip}\!\left(v_{i,t} + a_i^{v}\,\Delta v_{\max},\,
    v_{\min},\,v_{\max}\right),
\label{eq:v_update}\\[2pt]
\dot{\psi}_{i,t+1} \;&=\; \mathrm{clip}\!\left(\dot{\psi}_{i,t}
    + a_i^{\omega}\,\Delta\dot{\psi}_{\max},\;
    -\dot{\psi}_{\max}^{\star}(v),\,\dot{\psi}_{\max}^{\star}(v)\right),
\label{eq:omega_update}
\end{align}
with a speed-dependent turn-rate limit that enforces a minimum turn radius
$R_{\min}$,
\begin{equation}
\dot{\psi}_{\max}^{\star}(v) \;=\;
\min\!\left(\dot{\psi}_{\max},\; v / R_{\min}\right).
\label{eq:turn_limit}
\end{equation}
Heading and position are then propagated by explicit Euler integration with
step size $\Delta t$,
\begin{equation}
\psi_{i,t+1} = \left(\psi_{i,t} + \dot{\psi}_{i,t+1}\,\Delta t\right) \bmod 2\pi,
\qquad
\begin{bmatrix} x_{i,t+1}\\ y_{i,t+1}\end{bmatrix}
= \begin{bmatrix} x_{i,t}\\ y_{i,t}\end{bmatrix}
+ v_{i,t+1}\,\Delta t
  \begin{bmatrix}\cos\psi_{i,t+1}\\ \sin\psi_{i,t+1}\end{bmatrix}.
\label{eq:pose_update}
\end{equation}
Positions are clipped to $\mathcal{W}$. The action head is factorised, with
$a_i^{v}$ and $a_i^{\omega}$ each drawn from an independent categorical
distribution over $K=7$ bins mapping to multipliers
$c_k\in\{-1,-0.5,-0.1,0,+0.1,+0.5,+1\}$.
Parameters are shared within each role. Numerical values are listed in
Table~\ref{tab:kin_params}.

\section{Observation, Communication and Action Spaces}
\label{sec:obs_comm}

\paragraph{Sensing.}
Each agent has two range parameters: a sensing radius $R_{\mathrm{obs}}$
within which it directly detects other entities, and a communication radius
$R_{\mathrm{comm}}$ defining direct peer links. The direct neighbourhood and
the multi-hop communication subgroup of agent $i$ are
\begin{equation}
\mathcal{N}_i^{1} = \bigl\{ i' \in \mathcal{N}\setminus\{i\}\;\bigl|\;
 \rho_{ii'}\le R_{\mathrm{comm}} \bigr\},
\qquad
\mathcal{S}_i = \{i\}\cup \mathrm{TC}(\mathcal{N}_i^{1}),
\label{eq:comm_sets}
\end{equation}
where $\rho_{ab}=\|\mathbf{p}_a-\mathbf{p}_b\|_2$ and $\mathrm{TC}(\cdot)$ is
the transitive closure over the graph induced by $\mathcal{N}^{1}$. Members of
$\mathcal{S}_i$ pool their local detections, so the set of entities of
type $\xi\in\{\mathrm{ag},\mathrm{tg},\mathrm{rd}\}$ available to agent $i$
is
\begin{equation}
\mathcal{D}_i^{\xi} \;=\; \mathcal{K}^{\xi}\;\cup\;
\bigcup_{i'\in\mathcal{S}_i}\!\Bigl\{ e\in\xi\;\bigl|\;
\rho_{i'e}\le R_{\mathrm{obs}}\Bigr\},
\label{eq:shared_obs}
\end{equation}
where $\mathcal{K}^{\xi}\subseteq\xi$ denotes the subset of entities assumed
\emph{known} \emph{a priori} (e.g. briefed targets/radars); all other entities
must be detected through \eqref{eq:shared_obs}. The same multi-hop sharing
logic is used for the baseline and the FOFE encoder; the two methods differ
only in how $\mathcal{D}_i^{\xi}$ is embedded into the policy input.

\paragraph{Observation vector.}
Each agent's local observation decomposes into a fixed-size ego vector and
three typed entity channels,
\begin{equation}
o_i \;=\; \bigl(\, o_i^{\mathrm{self}},\;
\{o_{ii'}^{\mathrm{ag}}\}_{i'\in\mathcal{D}_i^{\mathrm{ag}}},\;
\{o_{ij}^{\mathrm{tg}}\}_{j\in\mathcal{D}_i^{\mathrm{tg}}},\;
\{o_{ik}^{\mathrm{rd}}\}_{k\in\mathcal{D}_i^{\mathrm{rd}}}\bigr).
\label{eq:obs_vector}
\end{equation}
The ego vector uses normalised kinematics and an alive flag
($d_{\mathrm{self}}=6$). Each entity feature contains relative range and
bearing expressed in the body frame of agent $i$ plus type-specific flags
($d_{\mathrm{ag}}=4,\,d_{\mathrm{tg}}=3,\,d_{\mathrm{rd}}=3$).

\paragraph{Strike and jam effects.}
A striker $i\in\mathcal{N}_{\mathrm{s}}$ has a forward engagement cone with
range $r^{\mathrm{Stk}}$ and half-angle $\tfrac{1}{2}\alpha^{\mathrm{Stk}}$;
the engaged target set is
\begin{equation}
\mathcal{A}_i^{\mathrm{Stk}} \;=\; \bigl\{ j\in\mathcal{T}\;\bigl|\;
\rho_{ij}\le r^{\mathrm{Stk}} \;\wedge\;
|\theta_{ij}|\le\tfrac{1}{2}\alpha^{\mathrm{Stk}}\bigr\},
\label{eq:strike_set}
\end{equation}
with $\theta_{ij}$ the body-frame bearing of $j$. Any $j\in\mathcal{A}_i^{\mathrm{Stk}}$
is removed from $\mathcal{T}$ deterministically. The jamming model is
environment-specific and is defined in Section~\ref{sec:envs}.

\section{Environment Models}
\label{sec:envs}

Two environments are used. Both share the kinematics, observation model,
communication graph and strike mechanics described above; they differ only in
how each radar's effective detection range is computed.

\subsection{Low-Fidelity Environment}
\label{sec:lf_env}

The low-fidelity (LF) environment models radar effectiveness by a single
binary jam state. The effective detection range of radar $k\in\mathcal{R}$ is
\begin{equation}
R^{\mathrm{eff}}_k \;=\;
\begin{cases}
\max\!\bigl(0,\;R_k - \Delta R\bigr), & \text{if } |\mathcal{N}_{\mathrm{j}}^{\mathrm{alive}}|\ge 1,\\[2pt]
R_k, & \text{otherwise,}
\end{cases}
\label{eq:lf_jam}
\end{equation}
with $R_k$ the nominal range and $\Delta R$ a fixed range reduction. An agent
$i$ is killed in a given step if
$\rho_{ik}\le R^{\mathrm{eff}}_k$ for any $k\in\mathcal{R}$, with a per-step
kill probability $p_{\mathrm{kill}}$. Equation~\eqref{eq:lf_jam} assumes
isotropic jamming, i.e.\ a single alive jammer suffices to suppress all
radars regardless of relative geometry. Low-fidelity parameters are
listed in Table~\ref{tab:lf_params}.

\subsection{High-Fidelity Environment}
\label{sec:hf_env}

The high-fidelity (HF) environment replaces the binary jam model with an
angular burn-through model based on the monostatic radar equation. The LF
kinematics, communication and reward structure are \emph{unchanged}; only the
mapping from radar/jammer geometry to $R^{\mathrm{eff}}$ is replaced. Three
effective ranges are defined per radar: the unconstrained range
$R_{\mathrm{unc}}$, the main-lobe burn-through range $R_{\mathrm{main}}$,
and the side-lobe burn-through range $R_{\mathrm{side}}$.

The unconstrained range follows from equating the received target echo power
to the receiver noise floor at a prescribed minimum detection SNR $\Gamma_{\min}$,
\begin{equation}
R_{\mathrm{unc}} \;=\;
\left(
\frac{P_t\,G_t\,G_r\,\lambda^{2}\,\sigma}
     {(4\pi)^{3}\, k\, T_{0}\, B_n\, L\, \Gamma_{\min}}
\right)^{1/4}.
\label{eq:R_unc}
\end{equation}
Jamming is modelled via the burn-through condition, where the received
jammer power equals the target echo at $R^{\mathrm{eff}}$. For main-lobe
jamming (the jammer lies inside the radar main-lobe sector) the effective
range reduces to
\begin{equation}
R_{\mathrm{main}} \;=\;
\left(
\frac{P_t\,G_t\,\sigma}{4\pi\,P_J\,G_J}\,R_{J}^{\,2}
\right)^{1/4},
\label{eq:R_main}
\end{equation}
and for side-lobe jamming, accounting for the reduced receive gain $G_{S}$
toward the jammer,
\begin{equation}
R_{\mathrm{side}} \;=\;
\left(
\frac{\sigma}{4\pi}\cdot
\frac{P_t\,G_t}{P_J\,G_J}\cdot
\frac{G_t\,R_{J}^{\,2}}{G_{S}}
\right)^{1/4}.
\label{eq:R_side}
\end{equation}
Here $P_t,P_J$ denote radar and jammer transmit powers, $G_t,G_r,G_J,G_{S}$
the corresponding antenna gains, $\lambda$ the wavelength, $\sigma$ the target
radar cross section, $R_J$ the radar--jammer range, $k$ the Boltzmann
constant, $T_0$ the system temperature, $B_n$ the receiver bandwidth and
$L$ the system losses. For a monostatic radar we use $G_r=G_t$.

Let $\theta_{kJ}$ denote the bearing of jammer $J$ relative to radar $k$, and
let $\theta_{\mathrm{main}},\theta_{\mathrm{side}}$ be the configured
main- and side-lobe half-widths. For any agent $i$ at bearing $\theta_{ki}$
from radar $k$, the per-agent, per-radar effective range is a piecewise
function of which jammer sector the agent lies in,
\begin{equation}
R^{\mathrm{eff}}_{ki} \;=\;
\begin{cases}
\min_{J\in\mathcal{N}_{\mathrm{j}}^{\mathrm{alive}}} R_{\mathrm{main}}(R_{kJ}), &
|\theta_{ki}-\theta_{kJ}|\le \theta_{\mathrm{main}} \;\text{for some }J,\\[2pt]
\min_{J\in\mathcal{N}_{\mathrm{j}}^{\mathrm{alive}}} R_{\mathrm{side}}(R_{kJ}), &
\theta_{\mathrm{main}} < |\theta_{ki}-\theta_{kJ}|\le \theta_{\mathrm{side}}
\;\text{for some }J,\\[2pt]
R_{\mathrm{unc}}, & \text{otherwise.}
\end{cases}
\label{eq:hf_eff}
\end{equation}
Kill mechanics are identical to the LF case but with the per-agent range
\eqref{eq:hf_eff}. Parameters are summarised in Table~\ref{tab:hf_params}.

\section{Reward Structure}
\label{sec:rewards}

Rewards are emitted per alive agent at every step and consist of sparse event
terms and dense shaping terms,
\begin{equation}
r_{i,t} \;=\; r_{i,t}^{\mathrm{sparse}} \;+\; r_{i,t}^{\mathrm{dense}},
\qquad
r_{i,t}^{\mathrm{dense}} = r^{\mathrm{time}} + r_{i,t}^{\mathrm{border}}
+ r_{i,t}^{\mathrm{radar}} + r_{i,t}^{\mathrm{app}} + r_{i,t}^{\mathrm{form}}
+ r_{i,t}^{\mathrm{ctrl}}.
\label{eq:reward_total}
\end{equation}

\paragraph{Sparse terms.}
Target kills, agent losses and mission completion use a convex combination of
a team-mean component and an individual credit,
\begin{equation}
r_i^{e} \;=\; \eta\,\bar R_e \;+\; (1-\eta)\,\tilde R_i^{e},\qquad e\in\{\mathrm{kill},\mathrm{loss},\mathrm{term}\},
\label{eq:sparse}
\end{equation}
with $\bar R_e$ the per-event reward averaged over alive agents and
$\tilde R_i^{e}$ the credit assigned to the individual responsible (the
striker whose cone \eqref{eq:strike_set} destroyed the target, or the
destroyed agent itself). The scalar $\eta\in[0,1]$ controls the team--
individual mix.

\paragraph{Dense terms.}
All dense terms share a common piecewise linear-exponential shape
$\phi_\xi(d;\boldsymbol{\theta}_\xi)$ parametrised by a weight pair
$(w^{\mathrm{lin}},w^{\mathrm{exp}})$, a knee $d^{\mathrm{knee}}$, an exponent
$\alpha$ and a cut-off $d^{\max}$,
\begin{equation}
\phi(d;\boldsymbol{\theta}) =
\begin{cases}
0, & d\ge d^{\max},\\[2pt]
w^{\mathrm{lin}}\,\dfrac{d^{\max}-d}{d^{\max}-d^{\mathrm{knee}}}, &
d^{\mathrm{knee}}\le d < d^{\max},\\[6pt]
w^{\mathrm{lin}} + w^{\mathrm{exp}}\Bigl(e^{\alpha(1-d/d^{\mathrm{knee}})}-1\Bigr), &
d < d^{\mathrm{knee}}.
\end{cases}
\label{eq:phi}
\end{equation}
Approach-type terms (striker$\to$target, jammer$\to$radar, formation) are
expressed as distance penalties and are zero at contact; avoidance-type terms
(border, radar-zone) are negative and increase as the agent enters the
hazard,
\begin{align}
r^{\mathrm{time}}_{i,t}   &= -c_{\mathrm{time}}, \\
r^{\mathrm{border}}_{i,t} &= -\phi\bigl(d^{\mathrm{edge}}_{i,t};\boldsymbol{\theta}_{\mathrm{bd}}\bigr), \\
r^{\mathrm{radar}}_{i,t}  &= -\!\!\sum_{k\in\mathcal{R}}\phi\bigl(d^{\mathrm{rd}}_{ik,t};\boldsymbol{\theta}_{\mathrm{rd}}\bigr), \\
r^{\mathrm{app}}_{i,t}    &= -\phi\bigl(d^{\mathrm{tg}}_{i,t};\boldsymbol{\theta}_{\mathrm{app}}\bigr),
    \quad i\in\mathcal{N}_{\mathrm{s}}, \\
r^{\mathrm{form}}_{i,t}   &= -\phi\bigl(d^{\mathrm{ally}}_{i,t};\boldsymbol{\theta}_{\mathrm{form}}\bigr),
    \quad i\in\mathcal{N}_{\mathrm{j}}, \\
r^{\mathrm{ctrl}}_{i,t}   &= -\lambda_{a}(a_i^{v})^{2} - \lambda_{\omega}(a_i^{\omega})^{2}.
\end{align}
$r^{\mathrm{time}}$ discourages long trajectories; $r^{\mathrm{border}}$
penalises proximity to the map boundary; $r^{\mathrm{radar}}$ penalises
proximity to radar threat zones; $r^{\mathrm{app}}$ pulls each striker toward
the nearest alive target; $r^{\mathrm{form}}$ keeps each jammer close to a
striker; $r^{\mathrm{ctrl}}$ penalises aggressive control inputs. Reward
parameters are listed in Table~\ref{tab:reward_params}.

\section{MAPPO Training}
\label{sec:mappo}

Both the baseline and the proposed method are trained with role-factorised
MAPPO~\textbf{[CITE: Yu et al., MAPPO, 2022]} built on the clipped PPO
objective~\textbf{[CITE: Schulman et al., PPO, 2017]}. Each role $\rho\in\{\mathrm{s},\mathrm{j}\}$
has its own actor $\pi_{\theta_\rho}$ and centralised critic $V_{\phi_\rho}$;
parameters are shared across all agents of the same role.

\paragraph{Advantage estimation.}
Rollouts of length $T$ are collected from $B$ parallel environments. For each
role, per-step TD residuals and generalised-advantage
estimates~\textbf{[CITE: Schulman et al., GAE, 2016]} are
\begin{equation}
\delta_t^{\rho} = r_t^{\rho}
+ \gamma V_{\phi_\rho}(s_{t+1})(1-d_t) - V_{\phi_\rho}(s_t),
\qquad
\hat{A}_t^{\rho} = \sum_{l=0}^{T-t-1}(\gamma\lambda)^{l}\,\delta_{t+l}^{\rho},
\label{eq:gae}
\end{equation}
with GAE return $\hat{R}_t^{\rho}=\hat{A}_t^{\rho}+V_{\phi_\rho}(s_t)$.
Within each minibatch, advantages are standardised:
$\tilde{A}=(\hat{A}-\mu_{\hat A})/\sigma_{\hat A}$.

\paragraph{Clipped policy loss.}
With importance ratio
$\varrho_t(\theta)=\exp(\log\pi_{\theta_\rho}(a_t\mid o_t) - \log\pi_{\theta_{\rho,\mathrm{old}}}(a_t\mid o_t))$,
\begin{equation}
\mathcal{L}^{\mathrm{PPO}}_{\rho}(\theta)
= -\,\mathbb{E}_t\!\Big[\min\!\bigl(\varrho_t\tilde{A}_t^{\rho},\;
\mathrm{clip}(\varrho_t,1-\epsilon,1+\epsilon)\tilde{A}_t^{\rho}\bigr)\Big]
- \beta\,\mathbb{E}_t\!\bigl[\mathcal{H}(\pi_{\theta_\rho})\bigr],
\label{eq:ppo_loss}
\end{equation}
and the critic loss is the unclipped mean-squared error
\begin{equation}
\mathcal{L}^{V}_{\rho}(\phi)
= \mathbb{E}_t\!\Big[\bigl(V_{\phi_\rho}(s_t) - \hat{R}_t^{\rho}\bigr)^{2}\Big].
\label{eq:val_loss}
\end{equation}
Rewards are rescaled by a running second moment (standard deviation only, no
mean subtraction) updated with a Welford estimator, preserving the sign of
sparse terminal events. Gradients are clipped per role at
$\|g\|_2\le g_{\max}$, and actor/critic are updated with independent Adam
optimisers.

\paragraph{MultiCategorical distribution.}
The factorised action distribution is the product of two independent
categoricals over the $K=7$ bins of $a_i^{v}$ and $a_i^{\omega}$. Logits are
sanitised by $\mathrm{clip}(\ell,-\ell_{\max},\ell_{\max})$ with
$\ell_{\max}=20$ to guard against occasional extreme values during early
exploration.

\section{Baseline: Flattened MAPPO}
\label{sec:baseline}

The baseline uses the same MAPPO objective, communication graph
\eqref{eq:comm_sets}--\eqref{eq:shared_obs}, environment and reward structure
as the proposed method, and differs only in how the variable-size entity sets
$\mathcal{D}_i^{\xi}$ in \eqref{eq:obs_vector} are encoded into a
fixed-dimensional input.

\paragraph{Observation flattening.}
For each entity type $\xi$ a fixed capacity $K_\xi$ is chosen. The entities
in $\mathcal{D}_i^{\xi}$ are sorted by range to agent $i$, truncated to the
first $K_\xi$, and padded with zeros when
$|\mathcal{D}_i^{\xi}|<K_\xi$. A per-slot visibility mask
$m_i^{\xi}\in\{0,1\}^{K_\xi}$ is concatenated with the feature block. The
resulting fixed-length observation
\begin{equation}
o_i^{\mathrm{flat}} \;=\; \Bigl[\, o_i^{\mathrm{self}} \,\|\,
\bigl(o_{ii'}^{\mathrm{ag}},m_{ii'}^{\mathrm{ag}}\bigr)_{i'=1}^{K_{\mathrm{ag}}} \,\|\,
\bigl(o_{ij}^{\mathrm{tg}},m_{ij}^{\mathrm{tg}}\bigr)_{j=1}^{K_{\mathrm{tg}}} \,\|\,
\bigl(o_{ik}^{\mathrm{rd}},m_{ik}^{\mathrm{rd}}\bigr)_{k=1}^{K_{\mathrm{rd}}} \,\Bigr],
\label{eq:flat_obs}
\end{equation}
has fixed dimension
$d_{\mathrm{in}} = d_{\mathrm{self}} + \sum_{\xi}(d_\xi+1)K_\xi$, so the
policy cannot generalise to observation sizes exceeding the training capacity.
The actor and centralised critic are realised as multi-layer perceptrons with
ReLU activations, depth $L$ and width $h$, parameter-shared within each role
(Table~\ref{tab:baseline_params}).

\begin{algorithm}[H]
\caption{Flattened MAPPO (one training iteration).}
\label{alg:flat_mappo}
\begin{algorithmic}[1]
\Require actors $\pi_{\theta_\rho}$, critics $V_{\phi_\rho}$, optimisers
  $\mathrm{Adam}_\rho^{\pi},\mathrm{Adam}_\rho^{V}$; hyperparameters
  $(B,T,E,M,\gamma,\lambda,\epsilon,\beta,g_{\max})$.
\For{$b=1,\dots,B$ in parallel}
  \State Reset environment $b$; set $\mathbf{s}^{(b)}_0\sim p_0$.
  \For{$t=0,\dots,T-1$}
    \State Compute $\mathcal{D}_i^{\xi}$ via
      \eqref{eq:comm_sets}--\eqref{eq:shared_obs}; build $o_i^{\mathrm{flat}}$
      via \eqref{eq:flat_obs} for each $i$.
    \State Sample $a_{i,t}\sim\pi_{\theta_\rho}(\cdot\mid o_{i,t}^{\mathrm{flat}})$.
    \State Step env; log $\bigl(s_t,o_{i,t}^{\mathrm{flat}},a_{i,t},r_{i,t},d_t,\log\pi_{\theta_\rho^{\mathrm{old}}}\bigr)$.
  \EndFor
\EndFor
\State For each role $\rho$, compute $V_{\phi_\rho}(s_t)$ and GAE
  $\hat{A}_t^{\rho},\hat{R}_t^{\rho}$ via \eqref{eq:gae}.
\For{$e=1,\dots,E$}
  \For{minibatch of size $M$}
    \State Standardise $\tilde A$; compute $\mathcal{L}^{\mathrm{PPO}}_{\rho}$
      \eqref{eq:ppo_loss} and $\mathcal{L}^{V}_{\rho}$ \eqref{eq:val_loss}.
    \State Backpropagate, clip $\|\nabla\|\le g_{\max}$, step $\mathrm{Adam}_\rho^{\pi},\mathrm{Adam}_\rho^{V}$.
  \EndFor
\EndFor
\end{algorithmic}
\end{algorithm}

\section{Proposed: FOFE-MAPPO}
\label{sec:fofe}

To make the policy agnostic to the number of visible teammates, targets and
radars, the flat encoding \eqref{eq:flat_obs} is replaced by a
permutation-invariant set encoder, Flexible Observation Feature Encoding
(FOFE), adapted from \textbf{[CITE: Wang et al., 2025/2026]}. The
environment, MAPPO objective and reward are unchanged; only the actor and
critic input pipelines are replaced.

\paragraph{Sequence-Exchangeable Encoding (SEE) layer.}
Let $X\in\mathbb{R}^{n\times d_{\mathrm{in}}}$ be the feature matrix of a
variable-size entity channel with visibility mask
$m\in\{0,1\}^{n}$. One SEE layer of hidden width $h$ computes
\begin{align}
\tilde X &= \mathrm{LayerNorm}(X), \label{eq:see_ln}\\
E &= \tilde X W_e,\qquad Z = \tilde X W_g,\qquad G = \mathrm{SiLU}(Z), \label{eq:see_proj}\\
e_{\mathrm{agg}} &= \max_{i:\,m_i=1}\, E_{i,:}\;\in\;\mathbb{R}^{h}, \label{eq:see_pool}\\
\mathrm{SEE}(X,m)_{i,:} &=
\bigl[\, G_{i,:}\odot E_{i,:}\,\|\,G_{i,:}\odot e_{\mathrm{agg}}\,\bigr]
\in\mathbb{R}^{2h}, \label{eq:see_cat}
\end{align}
with layer-norm omitted in the first SEE layer. Masked rows are zeroed after
concatenation. The left branch preserves per-entity local features; the
right branch broadcasts a gated global summary back to every row, yielding a
permutation-equivariant output.

\paragraph{FOFE block.}
A channel encoder stacks $L_{\mathrm{SEE}}$ SEE layers, applies a shared
per-entity MLP $f_\psi$, and collapses the set to a fixed-size vector by a
masked max-pool,
\begin{equation}
\mathrm{FOFE}(\mathcal{X},m) \;=\;
\max_{i:\,m_i=1}\, f_\psi\!\Bigl(\mathrm{SEE}_{L_{\mathrm{SEE}}}\!\circ\!\cdots\!\circ\!\mathrm{SEE}_{1}(X,m)\Bigr)_{\!i}
\;\in\;\mathbb{R}^{D_{\mathrm{FOFE}}}.
\label{eq:fofe}
\end{equation}
The output is permutation-invariant and independent of $n$. If no entity is
visible ($\sum_i m_i = 0$) the block returns a zero vector instead of
propagating $-\infty$ from the masked pool.

\paragraph{FOFE actor.}
For each role, the self-observation is mapped by an MLP $g_\xi$, and the
three entity channels are encoded by independent FOFE blocks. The resulting
vectors are concatenated and passed through a fusion MLP and a factorised
action head,
\begin{equation}
z_i \;=\; \bigl[\, g_\xi(o_i^{\mathrm{self}})\,\|\,
\mathrm{FOFE}_{\mathrm{ag}}\,\|\,
\mathrm{FOFE}_{\mathrm{tg}}\,\|\,
\mathrm{FOFE}_{\mathrm{rd}}\,\bigr],
\qquad
\boldsymbol{\ell}_i \;=\; W_{\mathrm{head}}\,\mathrm{FusionMLP}(z_i).
\label{eq:fofe_actor}
\end{equation}
The logits $\boldsymbol{\ell}_i\in\mathbb{R}^{2K}$ parametrise the
MultiCategorical distribution of Section~\ref{sec:mappo}.

\paragraph{FOFE critic.}
The critic receives the global state decomposed into three entity sets,
\emph{all} agents ($d{=}7$: position, speed, heading, $\omega$, role flag,
alive), all targets ($d{=}3$: position, alive), all radars ($d{=}4$:
position, active, $R^{\mathrm{eff}}$), plus a scalar episode-time feature
$\tau=t/T\in[0,1]$,
\begin{equation}
V_{\phi_\rho}(s) \;=\; W_v\,\mathrm{FusionMLP}\!\bigl(
\bigl[\mathrm{FOFE}^{c}_{\mathrm{ag}}(s)\,\|\,\mathrm{FOFE}^{c}_{\mathrm{tg}}(s)
\,\|\,\mathrm{FOFE}^{c}_{\mathrm{rd}}(s)\,\|\,\tau\bigr]\bigr).
\label{eq:fofe_critic}
\end{equation}
Each role owns a full set of three actor FOFE blocks and three critic FOFE
blocks; no weights are shared across roles. Network dimensions are listed in
Table~\ref{tab:fofe_params}.

\begin{algorithm}[H]
\caption{FOFE-MAPPO (one training iteration).}

\begin{algorithmic}[1]
\Require FOFE actors $\pi_{\theta_\rho}$ \eqref{eq:fofe_actor}, FOFE critics

    \State Backpropagate, clip $\|\nabla\|\le g_{\max}$, step

  \EndFor

\section{Performance Metrics}
\label{sec:kpis}

\subsection{Key Performance Indicators}
\label{sec:kpi_list}

Evaluation uses $N_{\mathrm{eval}}$ held-out layouts, unseen during training.
The primary KPIs are
\begin{enumerate}[leftmargin=1.25em]
\item \emph{Mission success rate} $\bar{s} = \tfrac{1}{N_{\mathrm{eval}}}
\sum \mathbb{1}[\,\text{all targets destroyed}\,]$;
\item \emph{Survival rate} per role, $\bar{\kappa}_\rho$;
\item \emph{Mean episode return} (per role and team);
\item \emph{Time-to-first-kill} and \emph{mission time} (steps until all
targets are destroyed or episode timeout);
\item \emph{Generalisation gap}: difference between $\bar{s}$ on
training layouts versus held-out layouts with $N_{\mathrm{s}},N_{\mathrm{j}},
N_{\mathrm{r}},N_{\mathrm{t}}$ outside the training distribution.
\end{enumerate}

\subsection{Emergent Tactics}
\label{sec:tactics}

Beyond aggregate KPIs, we analyse qualitative tactical behaviour through
spatial statistics of trajectories: (i) the \emph{suppress-then-penetrate}
index, defined as the fraction of successful missions in which at least one
jammer enters its suppression region before the earliest striker kill;
(ii) the \emph{formation coherence} $\bar d^{\mathrm{ally}}$, measured as the
mean cross-role distance during the approach phase; (iii) \emph{flanking
angle}, the bearing of the striker to the radar at the moment of the kill,
used to detect whether the policy learns to attack from the side-lobe sector
in the HF environment.
